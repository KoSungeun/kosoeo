CREATE TABLE ADMIN 
(
  NO NUMBER(4, 0) 
, MEMBERNO NUMBER(4, 0) 
) 

CREATE TABLE COMMENTS 
(
  NO NUMBER(4, 0) 
, MEMBERNO NUMBER(4, 0) 
, BOARDNO NUMBER(4, 0) 
, CONTENT VARCHAR2(200 BYTE) 
, COMMENTDATE DATE DEFAULT sysdate 
) 

CREATE TABLE THUMB 
(
  NO NUMBER(4, 0) 
, MEMBERNO NUMBER(4, 0) 
, BOARDNO NUMBER(4, 0) 
, UP NUMBER(1, 0) DEFAULT 0 
, DOWN NUMBER(1, 0) DEFAULT 0 
) 

CREATE TABLE FILES 
(
  NO NUMBER(4, 0) 
, BOARDNO NUMBER(4, 0) 
, PATH VARCHAR2(50 BYTE) 
, SUBMITNAME VARCHAR2(50 BYTE) 
, REALNAME VARCHAR2(50 BYTE) 
, CONTENTTYPE VARCHAR2(30 BYTE) 
, FILESIZE NUMBER(8, 0) 
, UPLOADDATE DATE DEFAULT sysdate 
)
